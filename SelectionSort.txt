#include <stdio.h>
int main()
{
    int i,a[10],min,j,n,temp;
     printf("enter array size: ");
   scanf("%d",&n);
   printf("enter array elements: ");
   for(i=0;i<n;i++)
        scanf("%d",&a[i]);
    printf("Before sorting:\n ");
    for(i=0;i<n;i++)
        printf("%d ",a[i]);
    
    for(i=0;i<n-1;i++){
        min=i;
        for(j=i+1;j<n;j++){
        if(a[min]>a[j]){
        min=j;
        }
            temp=a[min];
            a[min]=a[i];
            a[i]=temp;
        }
    }
        
    
     printf("\nAfter sorting:\n ");
    for(i=0;i<n;i++){
        printf("%d ",a[i]);
    }

    return 0;
}
